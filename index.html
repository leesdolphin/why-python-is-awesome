from contextlib import contextmanager

@contextmanager
def exit_on_error():
  try:
    yield  ## `with` block is executed here.
  except Exception:  ## `with` block raised an error.
    import sys
    sys.exit(1)

print("Hello, I'm going to divide by zero")
with exit_on_error:
  one_over_zero = 1 / 0
  ## Python exits instead of printing a stack trace.

ABC - as easy as 123

Writing a fully featured dictionary is really easy ... 

import collections.abc

class KeyDefaultDictionary(collections.abc.MutableMapping):

  def __init__(self):
    self.store = {}

  def __getitem__(self, key):
    if key not in self.store:
      self.store[key] = key
      return key
    return self.store[key]

  def __setitem__(self, key, value):
    self.store[key] = value

  def __delitem__(self, key, value):
    del self.store[key]

  def __iter__(self):
    return iter(self.store)

  def __len__(self):
    return len(self.store)
